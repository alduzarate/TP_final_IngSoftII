\begin{zed}
[NADADOR, ARBITRO, CARRERA]
\end{zed}

\begin{schema}{Torneo}
nadadoresInscriptos : CARRERA \pfun \power NADADOR \\
arbitrosAsignados : CARRERA \pfun \power ARBITRO
\end{schema}

\begin{schema}{TorneoInicial}
Torneo
\where
nadadoresInscriptos = \emptyset \\
arbitrosAsignados = \emptyset
\end{schema}

\begin{schema}{TorneoInv}
Torneo
\where
\dom(nadadoresInscriptos) \subseteq \dom(arbitrosAsignados)
\end{schema}

\begin{schema}{InscribirNadadorHayCompetidoresOk}
\Delta Torneo \\
n? : NADADOR \\
c? : CARRERA \\
\where
c? \in \dom(arbitrosAsignados) \\
c? \in \dom(nadadoresInscriptos) \\
n? \notin nadadoresInscriptos(c?) \\
nadadoresInscriptos' = nadadoresInscriptos \oplus \{c? \mapsto (nadadoresInscriptos(c?) \cup \{ n? \} ) \} \\
arbitrosAsignados' = arbitrosAsignados
\end{schema}

\begin{schema}{InscribirNadadorSinCompetidoresOk}
\Delta Torneo \\
n? : NADADOR \\
c? : CARRERA \\
\where
c? \in \dom(arbitrosAsignados) \\
c? \notin \dom(nadadoresInscriptos) \\
n? \notin nadadoresInscriptos(c?) \\
nadadoresInscriptos' = nadadoresInscriptos \cup \{c? \mapsto  \{ n? \} \} \\
arbitrosAsignados' = arbitrosAsignados
\end{schema}

\begin{schema}{NadadorYaInscripto}
\Xi Torneo \\
n? : NADADOR \\
c? : CARRERA \\
\where
c? \in \dom(arbitrosAsignados) \\
c? \in \dom(nadadoresInscriptos) \\
n? \in nadadoresInscriptos(c?) \\
nadadoresInscriptos' = nadadoresInscriptos \\
arbitrosAsignados' = arbitrosAsignados
\end{schema}

\begin{schema}{CarreraNoHabilitada}
\Xi Torneo \\
n? : NADADOR \\
c? : CARRERA \\
\where
c? \notin \dom(arbitrosAsignados) \\
c? \notin \dom(nadadoresInscriptos) \\
nadadoresInscriptos' = nadadoresInscriptos \\
arbitrosAsignados' = arbitrosAsignados
\end{schema}

\begin{zed}
InscribirNadador \defs InscribirNadadorHayCompetidoresOk \lor InscribirNadadorSinCompetidoresOk \lor NadadorYaInscripto \lor CarreraNoHabilitada
\end{zed}

\begin{schema}{AsignarArbitrosOk}
\Delta Torneo \\
ca? : \power ARBITRO \\
c? : CARRERA \\
\where
c? \notin \dom(arbitrosAsignados) \\
c? \notin \dom(nadadoresInscriptos) \\
nadadoresInscriptos' = nadadoresInscriptos \\
arbitrosAsignados' = arbitrosAsignados \cup \{c? \mapsto ca? \}
\end{schema}

\begin{schema}{CarreraYaHabilitada}
\Xi Torneo \\
ca? : \power ARBITRO \\
c? : CARRERA \\
\where
c? \in \dom(arbitrosAsignados) \\
nadadoresInscriptos' = nadadoresInscriptos \\
arbitrosAsignados' = arbitrosAsignados
\end{schema}

\begin{zed}
AsignarArbitros \defs AsignarArbitrosOk \lor CarreraYaHabilitada
\end{zed}

\begin{schema}{MostrarInfoCarreraHayInscriptosOk}
\Xi Torneo \\
c? : CARRERA \\
ni! : \power NADADOR \\
aa! : \power ARBITRO \\
\where
c? \in \dom(arbitrosAsignados) \\
c? \in \dom(nadadoresInscriptos) \\
ni! = nadadoresInscriptos(c?) \\
aa! = arbitrosAsignados(c?)
\end{schema}

\begin{schema}{MostrarInfoCarreraSinInscriptosOk}
\Xi Torneo \\
c? : CARRERA \\
ni! : \power NADADOR \\
aa! : \power ARBITRO \\
\where
c? \in \dom(arbitrosAsignados) \\
c? \notin \dom(nadadoresInscriptos) \\
ni! = \emptyset \\
aa! = arbitrosAsignados(c?)
\end{schema}

\begin{schema}{NoHayInfoCarrera}
\Xi Torneo \\
c? : CARRERA \\
ni! : \power NADADOR \\
aa! : \power ARBITRO \\
\where
c? \notin \dom(arbitrosAsignados) \\
c? \notin \dom(nadadoresInscriptos) \\
ni! = \emptyset \\
aa! = \emptyset
\end{schema}

\begin{zed}
MostrarInfoCarrera \defs MostrarInfoCarreraHayInscriptosOk \lor MostrarInfoCarreraSinInscriptosOk \lor NoHayInfoCarrera
\end{zed}
